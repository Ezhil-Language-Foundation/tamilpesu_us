from .ipaconvert import ipa

[docs]def txt2ipa(text, broad=True):
    
    lat = tam2lat(text)
    lat = " " + lat + " "
    ipa_text = ipa(lat)
    
    # make memory free 
    del lat 
    
    if broad: ipa_text = broad_ipa(ipa_text)
    
    return ipa_text
